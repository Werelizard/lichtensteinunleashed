LUC_SupportCalculateContributions = {
	set_variable = {
		LUC_SupportDailyChange_var = 0

		LUC_SupportLiberal_var = LUC_SupportLiberalInfluence_var
	}
	divide_variable = {
		LUC_SupportLiberal_var = 100
	}
	if = { # If disloyaly
		limit = {
			NOT = {
				check_variable = {
					LUC_SupportLiberalLoyalty_var > 30
				}
			}
		}
		multiply_variable = {
			LUC_SupportLiberal_var = -1
		}

			LUC_SupportDailyChange_var = LUC_SupportLiberal_var
		}

			LUC_SupportDailyChange_var = LUC_SupportLiberal_var

LUC_UpdateLandtagModifier = {
	# Clears all modifiers
	remove_ideas = {
		LUC_LandtagVeto
		LUC_LandtagOpposition
		LUC_LandtagNeutral
		LUC_LandtagSupport
		LUC_LandtagControl
	}
	if = {
		limit = {
			NOT = {
				check_variable = {
					LUC_Confidence_var > 10
				}
			}
		}
		add_ideas = LUC_LandtagVeto
	}
	else_if = {
		limit = {
			NOT = {
				check_variable = {
					LUC_Confidence_var > 40
				}
			}
			check_variable = {
				LUC_Confidence_var > 10
			}
		}
		add_ideas = LUC_LandtagOpposition
	}
	else_if = {
		limit = {
			NOT = {
				check_variable = {
					LUC_Confidence_var < 60
				}
			}
			check_variable = {
				LUC_Confidence_var < 90
			}
		}
		add_ideas = LUC_LandtagSupport
	}
	else_if = {
		limit = {
			NOT = {
				check_variable = {
					LUC_Confidence_var < 90
				}
			}
		}
		add_ideas = LUC_LandtagControl
	}
	else = {
		add_ideas = LUC_LandtagNeutral
	}
}

LUC_DisableLandtag = {
	remove_ideas = {
		LUC_LandtagVeto
		LUC_LandtagOpposition
		LUC_LandtagNeutral
		LUC_LandtagSupport
		LUC_LandtagControl
	}
	clr_country_flag = LUC_EnableLandtag_flag
}
