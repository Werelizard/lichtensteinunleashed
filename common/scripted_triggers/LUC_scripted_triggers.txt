LUC_check_ruling_party_is_50 = {
	if = {
		limit = {
			has_government = democratic
		}
		NOT = {
			democratic < 0.5
		}
	}
	else_if = {
		limit = {
			has_government = communism
		}
		NOT = {
			communism < 0.5
		}
	}
	else_if = {
		limit = {
			has_government = fascism
		}
		NOT = {
			fascism < 0.5
		}
	}
	else = {
		NOT = {
			neutrality < 0.5
		}
	}
}
