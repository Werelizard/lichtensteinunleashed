defined_text = { # Landtag Support
	name = LUC_GetLandtagState
	text = {
		trigger = {
			NOT = {
				check_variable = {
					SUS.LUC_Confidence_var < 90
				}
			}
		}
		localization_key = "Total Landtag Support"
	}
	text = {
		trigger = {
			NOT = {
				check_variable = {
					SUS.LUC_Confidence_var < 60
				}
			}
			check_variable = {
				SUS.LUC_Confidence_var < 90
			}
		}
		localization_key = "Supportive Landtag"
	}
	text = {
		trigger = {
			check_variable = {
				SUS.LUC_Confidence_var > 40
			}
			check_variable = {
				SUS.LUC_Confidence_var < 60
			}
		}
		localization_key = "Indifferent Landtag"
	}
	text = {
		trigger = {
			NOT = {
				check_variable = {
					SUS.LUC_Confidence_var > 40
				}
			}
			check_variable = {
				SUS.LUC_Confidence_var > 10
			}
		}
		localization_key = "Landtag Opposition"
	}
	text = {
		trigger = {
			NOT = {
				check_variable = {
					SUS.LUC_Confidence_var > 10
				}
			}
		}
		localization_key = "Landtag Veto"
	}
	text = {
		localization_key = " "
	}
}
