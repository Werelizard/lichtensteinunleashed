scripted_gui = {
	LUC_GOVERNMENTCONFIDENCE = {
		window_name = LUC_GovernmentConfidence_ui_window
		context_type = decision_category

		dirty = SUS.LUC_Confidence_var

		properties = {
			confidence_bar_fill = {
				frame = SUS.LUC_Confidence_var
			}
		}
		triggers = {
			# Veto range
			confidence_bar_frame_1_visible = {
				check_variable = {
					LUC_Confidence_var > 10
				}
			}
			confidence_bar_frame_1_active_visible = {
				NOT = {
					check_variable = {
						LUC_Confidence_var > 10
					}
				}
			}
			# Oppositon range
			confidence_bar_frame_2_visible = {
				OR = {
					NOT = {
						check_variable = {
							LUC_Confidence_var > 10
						}
					}
					check_variable = {
						LUC_Confidence_var > 40
					}
				}
			}
			confidence_bar_frame_2_active_visible = {
				check_variable = {
					LUC_Confidence_var > 10
				}
				NOT = {
					check_variable = {
						LUC_Confidence_var > 40
					}
				}
			}
			# Neutral range
			confidence_bar_frame_3_visible = {
				OR = {
					NOT = {
						check_variable = {
							LUC_Confidence_var < 60
						}
					}
					NOT = {
						check_variable = {
							LUC_Confidence_var > 40
						}
					}
				}
			}
			confidence_bar_frame_3_active_visible = {
				check_variable = {
					LUC_Confidence_var < 60
				}
				check_variable = {
					LUC_Confidence_var > 40
				}
			}
			# Support range
			confidence_bar_frame_4_visible = {
				OR = {
					check_variable = {
						LUC_Confidence_var < 60
					}
					NOT = {
						check_variable = {
							LUC_Confidence_var < 90
						}
					}
				}
			}
			confidence_bar_frame_4_active_visible = {
				check_variable = {
					LUC_Confidence_var < 90
				}
				NOT = {
					check_variable = {
						LUC_Confidence_var < 60
					}
				}
			}
			# Total control range
			confidence_bar_frame_5_visible = {
				check_variable = {
					LUC_Confidence_var < 90
				}
			}
			confidence_bar_frame_5_active_visible = {
				NOT = {
					check_variable = {
						LUC_Confidence_var < 90
					}
				}
			}
		}
	}
}
